import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Loading data
data = pd.read_csv('dataset_space.csv') 

# Basic statistical analysis
print(data.describe())

# Make sure the Date column is of datetime type
data['Date'] = pd.to_datetime(data['Date'], errors='coerce')

# Distribution Visualisation
# View the distribution of PayloadMass
plt.figure(figsize=(10, 6))
sns.histplot(data['PayloadMass'], kde=True)
plt.title('Payload Mass Distribution')
plt.xlabel('Payload Mass (kg)')
plt.ylabel('Frequency')
plt.show()

# Disaggregated data analysis
# Analyse the frequency of LaunchSite
launch_site_counts = data['LaunchSite'].value_counts()
print(launch_site_counts)
launch_site_counts.plot(kind='bar')
plt.title('Launch Site Frequency')
plt.xlabel('Launch Site')
plt.ylabel('Counts')
plt.show()

# relevance analysis
# View correlation between numeric data
plt.figure(figsize=(10, 8))
sns.heatmap(data.corr(), annot=True, cmap='coolwarm')
plt.title('Correlation Matrix')
plt.show()


# time series analysis
# View the number of flights over time
data['Year'] = data['Date'].dt.year
flights_per_year = data.groupby('Year')['FlightNumber'].count()
flights_per_year.plot(kind='line')
plt.title('Flights per Year')
plt.xlabel('Year')
plt.ylabel('Number of Flights')
plt.show()
